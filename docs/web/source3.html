10. 질의응답 창을 통해 포털 검색 사이트 이름을 입력 받고, switch 문을 이용하여 입력한 이름에 해당하는 사이트로 이동하도록 작성

let url

   break;
   case "다음" : url = "www.daum.net";
   break;
   case "구글" : url = "www.google.com";
   break;
   default: alert("보기 중에 없는 사이트입니다!");
}
if(url) location.href = "http://" + url;
